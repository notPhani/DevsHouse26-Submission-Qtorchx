Benchmarks & Integration / Section 04

From Field to Circuit.

Drop-in PyTorch API. Validated against Qiskit Aer on Bell, GHZ, and QFT benchmarks.

Python API

# 1. Build the circuit topology
import torch
from qtorchx.core.primitives import Circuit, Gate
from qtorchx.noise.qnaf import PhiManifoldExtractor
from qtorchx.core.backend import QtorchBackend

circuit = Circuit(num_qubits=128)
circuit.add(Gate("H", qubits=[0], t=0))
circuit.add(Gate("CNOT", qubits=[0, 1], t=1))

# 2. Define a Learnable Decoherence Map (DPM)
# Maps the 6D latent Φ-field to Pauli error probabilities.
DPM = torch.randn(3, 6, device="cuda", requires_grad=True)
BPO = torch.zeros(3, device="cuda", requires_grad=True)

# 3. Extract the Continuous Spatiotemporal Φ-Field
extractor = PhiManifoldExtractor(circuit, DPM, BPO, device="cuda")
phi_manifold = extractor.GetManifold()  # Shape: (6, 128, T)
noisy_circuit = extractor.annotate_circuit()

# 4. Execute GPU-Accelerated Simulation
backend = QtorchBackend(circuit=noisy_circuit, simulate_with_noise=True)
statevector = backend.get_final_statevector()

# 5. Gradients flow natively through the noise manifold!
loss = compute_fidelity(statevector, target_state)
loss.backward()

print(f"∇ DPM gradients computed: {DPM.grad is not None}")
